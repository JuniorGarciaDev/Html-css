DOCTYPE HTML = INICIO DO CODIGO (INFORMA QUE ISSO E UM DOCUMENTO HTML)
HTML = AQUI É AONDE FICA OS COMANDO BODY,HEAA (RESUMINDO TUDO FICA DENTRO DESSA TAG MENOS O DOCTYPE HTML)
BODY = CORPO DO HTML (PARAGRO TITULO E TALS)
HEAD = CABEÇA DO HTML (CONFIGURAÇOES)
HIERARQUIA DE TITULO 
(
H1 = Titulo
H2 = SUB TITULO DE H1
H3 = SUB TITULO DE H2
H4 = SUB TITULO DE H3
H5 = SUB TITULO DE H4
H6 = SUB TITULO DE H5(SO VAI DO H1 ATE O H6 NAO TEM MAIS)
)

P = PARAGRAFO

LANG = LINGUAGEM

UTF-8 = PERMITI USO DE ACENTUAÇAO

HR = CRIA UMA LINHA HORIZONTAL NO SITE EX: __________

&LT = USADO PARA USAR SINAL DE MENOR < QUANDO NAO É UM COMANDO E SIM ALGO PARA ESCREVER
&GT = USANDO PARA USAR SINAL DE MAIOR > (MESMA EXPLICAÇAO DE CIMA)

&#X = USADO PARA BOTAR EMOJIS EX: &#x1F42F (RETIRA O UT- DO EMOJI E BOTAR NO LUGAR &#X)

IMG = PARA BOTAR IMAGEM NO SITE

BR = QUEBRA DE LINHA

STRONG = DEIXA EM NEGRITO OU COR MAIS FORTE
EM = DEIXA EM ITALICO OU DA ENFASE
MARK = DEIXA MARCADO OU DEIXA UMA COR DE FUNDO DIFERENTE

BIG = DEIXA A LETRA GRANDE (OBSOLENTA)
SMALL = DEILA LETRA PEQUENA

DEL = TEXTO DELETADO (deixa uma linha no meio do texto)
INS = TEXTO INSERIDO (DEIXA UMA LINHA EMBAIXO DO TEXTO)

SUP = SOBESCRITO (VEJA MAIS NO EX008)
SUB = SUBSCRITO  (VEJA MAIS NO EX008)

CODE = FONTE MENOESPAÇADA (MOSTRA TIPO UM PARAGRAFO SO QUE COM UMA FONTE QUE PAREÇA UM CODIGO)
PRE = VOCE CONSEGUE USAR INDENTAÇAO NO SEU CODIGO(PODE MESCLAR O PRE COM O CODE)

Q = ASPAS ,USE QUANDO PRECISAR ESCREVER UMA CITAÇAO

BLOCKQUOTE = DEIXA O TEXTO AFASTADO DA MARGEM

BLOCKQUOTE CITE = CITA UM LINK
ABBR = ABREVIAÇAO

OL=INICIA UMA LISTA ORDENADA (A, 1, NUMEROS ROMANOS E TALS)

UL = INICIA UMA LISTA NAO ORDENADA (QUADRADO REDONDO E TALS )

DL = INICIA UMA LISTA DE DEFINIÇÃO (E COMO SE FOSSE UM DICIONARIO EU VOU BOTAR UM TERMO E UMA DISCRIÇAO)

DT = TERMO	(DO DL)
DD = DISCRIÇAO (DO DL)

LI = MOSTRA UM NUMERO OU LETRA ( SE A LISTA FOR ORDENADA)
LI = MOSTRA QUADRADO OU REDONDO (SE A LISTA NAO FOR ORDENADA)

TYPE = (CONFIGURA AS LISTA EX: OL TYPE"A" MOSTRA UMA LETRA)
TYPE = (CONFIGURA AS LISTA EX: UL TYPE"CIRCLE" MOSTRA UM CIRCULO)
DENTRO DO TYPE SE USA
TYPE DO UL
( 
CIRCLE = MOSTRA UMA BOLA COM UMA BORDA PRETA SEM PREENCHIMENTO
DISC = PADRÃO. MOSTRA UMA BOLA TOTALMENTE PINTADA
SQUARE = MOSTRA UM QUADRADO TOTALMENTE PINTADO
)

TYPE DO OL
(
1 - VALOR PADRAO CRIA LISTA NUMERADA. EX:1,2,3,...
A = CRIA LISTA ALFABÉTICAS EM MAIÚSCULAS. EX: A,B,C,...
a = CRIA LISTA ALFABÉTICAS EM MINÚSCULAS. EX:a,b,c,...
I = CRIA LISTA COM ALGARISMOS ROMANOS EM MAIÚSCULAS EX: I,II,III,IV,...
i = CRIA LISTA COM ALGARISMO ROMANOS EM MINÚSCULAS, EX:i,ii,iii,iv,....
)

A = PODE BOTAR LINK NO SITE
HREF = AONDE BOTA O LINK EX: <a href = "LINK">
HREFLANG (INDICA IDIOMA PRINCIPAL DO SITE)
USANDO TAG TARGET EX:TARGET ="_BLANK" 
(
_BLANK = VAI ABRIR LINK EM UMA NOVA JANELA
_SELF = VAI ABRIR LINK NA JELA ATUAL
_TOP = VAI DESFAZER TODOS OS FRAMES E ABRIR NO DESTINO NO NAVEGADOR COMPLETO( EU NAO USO )
_PARENT SIMILIA AO TOP EM UMA REFERENCIA A JANELA MAE 
)

USANDO TAG REL EX REL="EXTERNAL"
(
NEXT = INDICA QUE O LINK É PARA PROXIMA PAGINA DO DOCUMENTO ATUAL
PREV = INDICA QUE O LINK É PARA A PAGINA ANTERIOR DO DOCUMENTO ATUAL
AUTHOR = INDICA QUE É UM LINK PARA O SITE DO AUTOR
EXTERNAL = INDICA QUE É UM LINK PARA OUTRO SITE
NOGOLLOW = INDICA QUE É UM LINK PARA UM SITE NAO ENDOSSADO,COMO LINK PAGO
)

USANDO TAG DOWNLOAD EX A HREF="ARQUIVO/MEULIVRO.PDF" DOWNLOAD="MEULIVRO.PDF" TYPE="APLICATION/PDF"
(
DOWNLOAD = PARA O LINK FAZER DOWNLOAD
TYPE = O TIPO DO ARQUIVO QUE VAI FAZER DOWNLOAD (PODE SER PDF,MP4,ZIP E OUTROS)
)


ÁUDIO = BOTA AUDIO NO SEU SITE  EX: <audio preload ="metadata" controls autoplay loop"
(
PRELOAD = INDICA SE O ÁUDIO SERÁ PRÉ-CARREGADO OU NAO ACEITA 3 VALORES
METADATA = vai carregar apenas as informações sobre o arquivo (tamanho,
tempo, informações de direitos, etc) 
AUTO = (padrão) vai carregar o arquivo de áudio inteiro assim que a página for
carregada, mesmo que o usuário nunca aperte o play 
NONE = none não vai carregar absolutamente nada até que o usuário clique no botão
play ou um script inicie a reprodução
(

VIDEO = BOTA VIDEO NO SEU SITE
(
WIDTH = vai indicar a largura que o vídeo vai ter na tela. Nesse exemplo, 600px.
POSTER= configura uma imagem que vai aparecer como uma capa, enquanto o 
visitante não aperta o play para reproduzir o vídeo 
)

CONTROLS = ADICIONA UM BOTAO PARA INICIA UM VIDEO/AUDIO
AUTOPLAY = O VIDEO/AUDIO INICIA SOZINHO (NAO PARECE FUCIONA NO CHROME)
LOOP = SEMPRE QUE O VIDEO/AUDIO ACABA ELE INICIARA NOVAMENTE
TYPE = TIPO DO ARQUIOVO DO AUDIO/VIDEO
SOURCE = AONDE BOTA O TYPE E O ARQUIVO DO AUDIO/VIDEO EX: SOURCE SRC= "VIDEO/AUDIO" TYPE="TIPOE DO AUDIO/VIDEO")
